JavaScript 核心语法学习

基础语法演示


// 变量声明与类型检测
let a = 100;
const api_key = 'ssssx-=-dddd';
console.log(`变量a的值：${a}, 类型：${typeof a}`);
console.log(`常量api_key的值：${api_key}, 类型：${typeof api_key}`);

// 类型转换与比较
let b = '1';
console.log(`字符串与数字相加：${a + b}`);
console.log(`类型转换后相加：${a + Number(b)}`);

// 严格比较
let c = '100';
console.log(`== 比较结果：${a == c}`);
console.log(`=== 比较结果：${a === c}`);

// 条件语句
let current_time = '12:20';
if (current_time == '12:20') {
    console.log('冲啊！');
} else {
    console.log('假装学习……');
}

// 循环语句
console.log('for循环输出：');
for (let i = 1; i <= 10; i++) {
    console.log(i);
}

console.log('while循环输出：');
let hug = 60;
while (hug > 0) {
    console.log('继续干饭……');
    hug -= 10;
}

// 函数定义与使用
function intro() {
    console.log('你好，我是一个脚本！');
}
intro();

function intro2(content) {
    console.log(`以下是我介绍的内容：${String(content)}`);
}
intro2('这是传入的参数');

function intro3(content1, content2) {
    console.log(`以下是我介绍的内容：${String(content1)} ${String(content2)}`);
}
intro3('参数1', '参数2');

function intro4(content) {
    console.log(content);
    return content + '这是返回值';
}
let result = intro4('带返回值的函数');
console.log(`函数返回值：${result}`);
